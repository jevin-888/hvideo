  41:                              int32_t format, int32_t deviceId) {
  42: #ifdef __ANDROID__
  43:   LOG_DEBUG("[AudioPlayer] initialize started: sampleRate=%d, channels=%d, format=%d",
  44:            sampleRate, channelCount, format);
  45:
  46:   shutdown();
  47:

  50:   channelCount_ = channelCount;
  51:   format_ = format;
  52:   deviceId_ = deviceId;
  53:
  54:   // Calculate bytes per frame
  55:   if (format == AAUDIO_FORMAT_PCM_I16) {
  56:     bytesPerFrame_ = channelCount_ * sizeof(int16_t);
  57:   } else if (format == AAUDIO_FORMAT_PCM_FLOAT) {
  58:     bytesPerFrame_ = channelCount_ * sizeof(float);
  59:   } else {
  60:     LOG_ERROR("[AudioPlayer] Unsupported audio format: %d", format);
  61:     return false;
  62:   }
  63:
  64:   LOG_DEBUG("[AudioPlayer] bytesPerFrame=%d", bytesPerFrame_);
  65:
  66:   // Create AAudio stream builder
  67:   AAudioStreamBuilder *builder = nullptr;
  68:   aaudio_result_t result = AAudio_createStreamBuilder(&builder);
  69:   if (result != AAUDIO_OK) {
  70:     LOG_ERROR("[AudioPlayer] Failed to create AAudio stream builder: %d", result);
  71:     return false;
  72:   }
  73:
  74:   LOG_DEBUG("[AudioPlayer] AAudio stream builder created successfully");
  75:
  76:   // Configure stream parameters
  77:   AAudioStreamBuilder_setDirection(builder, AAUDIO_DIRECTION_OUTPUT);
  78:   AAudioStreamBuilder_setPerformanceMode(builder,
  79:                                          AAUDIO_PERFORMANCE_MODE_LOW_LATENCY);
  80:   AAudioStreamBuilder_setFormat(builder, format_);
  81:   AAudioStreamBuilder_setChannelCount(builder, channelCount_);
  82:   AAudioStreamBuilder_setSampleRate(builder, sampleRate_);
  83:

  85:   if (deviceId_ != AAUDIO_UNSPECIFIED) {
  86:     AAudioStreamBuilder_setDeviceId(builder, deviceId_);
  87:     LOG_DEBUG("Setting audio device ID: %d", deviceId_);

  89:
  90:   // Note: setUsage and setContentType require API 28+, current min SDK is 26
  91:   // If these features are needed, use conditional compilation:
  92:   // #if __ANDROID_API__ >= 28

  94:   //     AAudioStreamBuilder_setContentType(builder, AAUDIO_CONTENT_TYPE_MUSIC);

  98:   AAudioStreamBuilder_setErrorCallback(builder, errorCallback, this);

 103:   AAudioStreamBuilder_setSharingMode(builder, AAUDIO_SHARING_MODE_EXCLUSIVE);

 123:   // 会从 0 直接跳到全幅 → click。和 flush() 走同一套防爆音路径。
 124:   currentVolume_.store(0.0f);
 125:   targetVolume_.store(0.0f);
 126:   volumeStep_.store(0.0f);
 127:   fadeComplete_.store(true);
 128:   pendingFadeInAfterFlush_.store(true, std::memory_order_release);
 129:   LOG_INFO("[AudioPlayer] Initial volume zeroed, fade-in armed for first write");
 130:
 131:   // 拉高 buffer size 消除启动期 underrun：
 132:   // EXCLUSIVE + LOW_LATENCY 默认 bufferSize = 2 * framesPerBurst (~4-8ms)，
 133:   // 在首几包音频写入未稳定前极易 underrun（起播连续 3-5 次 click/断续）。
 134:   // 把 bufferSize 拉到 4 * framesPerBurst (~16-32ms)，在 bufferCapacity 范围内。
 135:   {
 136:     int32_t framesPerBurst = AAudioStream_getFramesPerBurst(stream_);
 137:     int32_t bufferCapacity = AAudioStream_getBufferCapacityInFrames(stream_);
 138:     int32_t currentBufferSize = AAudioStream_getBufferSizeInFrames(stream_);
 139:     if (framesPerBurst > 0 && bufferCapacity > 0) {
 140:       int32_t desired = framesPerBurst * 12;
 141:       if (desired > bufferCapacity) desired = bufferCapacity;
 142:       if (desired > currentBufferSize) {
 143:         aaudio_result_t r = AAudioStream_setBufferSizeInFrames(stream_, desired);
 144:         if (r >= 0) {
 145:           LOG_INFO("[AudioPlayer] AAudio buffer resized: %d -> %d (burst=%d capacity=%d)",
 146:                    currentBufferSize, r, framesPerBurst, bufferCapacity);
 147:         } else {
 148:           LOG_WARN("[AudioPlayer] AAudio setBufferSizeInFrames failed: %d (keep default %d)",
 149:                    r, currentBufferSize);
 150:         }
 151:       } else {
 152:         LOG_INFO("[AudioPlayer] AAudio buffer kept: size=%d burst=%d capacity=%d",
 153:                  currentBufferSize, framesPerBurst, bufferCapacity);
 154:       }
 155:     }
 156:   }
 157:
 158:   // Verify actual parameters
 159:   int32_t actualSampleRate = AAudioStream_getSampleRate(stream_);
 160:   int32_t actualChannelCount = AAudioStream_getChannelCount(stream_);
 161:   int32_t actualFormat = AAudioStream_getFormat(stream_);
 162:   int32_t actualDeviceId = AAudioStream_getDeviceId(stream_);
 163:
 164:   LOG_INFO("[AudioPlayer] AAudio stream opened: requested=%dHz/%dch/fmt%d actual=%dHz/%dch/fmt%d deviceId=%d",
 165:            sampleRate_, channelCount_, format_,
 166:            actualSampleRate, actualChannelCount, actualFormat, actualDeviceId);
 167:
 168:   // Defensive check: if actual format does not match requested format,
 169:   // close the stream to avoid interpreting audio data with wrong sample format.
 170:   if (actualFormat != format_) {
 171:     LOG_ERROR("[AudioPlayer] Actual audio format (%d) != requested format (%d), "
 172:               "closing stream", actualFormat, format_);
 173:     AAudioStream_close(stream_);
 174:     stream_ = nullptr;
 175:     AAudioStreamBuilder_delete(builder);
 176:     return false;
 177:   }
 178:
 179:   if (actualChannelCount != channelCount_) {
 180:     LOG_ERROR("[AudioPlayer] Actual channel count (%d) != requested channel count (%d), closing stream",
 181:               actualChannelCount, channelCount_);
 182:     AAudioStream_close(stream_);
 183:     stream_ = nullptr;
 184:     AAudioStreamBuilder_delete(builder);
 185:     return false;
 186:   }
 187:
 188:   if (actualSampleRate > 0 && actualSampleRate != sampleRate_) {
 189:     LOG_WARN("[AudioPlayer] Actual sample rate differs from requested: requested=%d actual=%d",
 190:              sampleRate_, actualSampleRate);
 191:   }
 192:
 193:   // Cleanup builder
 194:   AAudioStreamBuilder_delete(builder);
 195:
 196:   return true;
 197: #else
 198:   return false;
 199: #endif

 382: void AudioPlayer::flush() {
 383: #ifdef __ANDROID__
 384:   // Clear audio queue to prevent residual data causing audio tail
 385:   {
 386:     std::lock_guard<std::mutex> lock(queueMutex_);
 387:     while (!audioQueue_.empty()) {
 388:       AudioQueueItem item = audioQueue_.front();
 389:       audioQueue_.pop();
 390:       delete[] item.data;
 391:     }
 392:   }
 393:   // 切歌/音轨切换防爆音：把当前音量强制归零，并标记"下次 write() 自动淡入"。
 394:   // 这样上层无论淡出是否完成、淡入是否按时调用，新歌首批样本一定从 0 开始爬升。
 395:   currentVolume_.store(0.0f);

 397:   volumeStep_.store(0.0f);
 398:   fadeComplete_.store(true);  // 保持当前 0 音量，直到 write() 触发淡入
 399:   pendingFadeInAfterFlush_.store(true, std::memory_order_release);
 400:   LOG_DEBUG("AudioPlayer queue flushed; volume zeroed, fade-in armed for next write");

 454:   {

 458:              shuttingDown_.load(std::memory_order_acquire);

 461:       aaudio_stream_state_t state = stream_ ? AAudioStream_getState(stream_) : AAUDIO_STREAM_STATE_UNINITIALIZED;
 462:       if (!shuttingDown_.load(std::memory_order_acquire) &&
 463:           (state == AAUDIO_STREAM_STATE_STARTED || state == AAUDIO_STREAM_STATE_STARTING)) {
 464:         playing_.store(true, std::memory_order_release);
 465:       } else {
 466:         static int dropCount = 0;
 467:         if (dropCount++ % 200 == 0) {
 468:           LOG_WARN("AudioPlayer::write: queue full or stopped, dropping audio data (queue=%zu playing=%d shutdown=%d state=%d)",
 469:                    audioQueue_.size(), (int)playing_.load(std::memory_order_acquire),
 470:                    (int)shuttingDown_.load(std::memory_order_acquire), state);
 471:         }
 472:         return 0;
 473:       }
 474:     }
 475:     if (!hasSpace && audioQueue_.size() >= MAX_QUEUE_SIZE) {
 476:       static int queuePressureCount = 0;
 477:       if (queuePressureCount++ % 50 == 0) {
 478:         aaudio_stream_state_t state = stream_ ? AAudioStream_getState(stream_) : AAUDIO_STREAM_STATE_UNINITIALIZED;
 479:         LOG_WARN("AudioPlayer::write: queue pressure, dropping oldest queued audio block (queue=%zu max=%zu state=%d)",
 480:                  audioQueue_.size(), MAX_QUEUE_SIZE, state);
 481:       }
 482:       AudioQueueItem &oldest = audioQueue_.front();
 483:       if (oldest.data) {
 484:         delete[] oldest.data;
 485:         oldest.data = nullptr;
 486:       }
 487:       audioQueue_.pop();
 488:     }
 489:   }
 490:
 491:   // Allocate buffer and copy data
 492:   int32_t dataSize = numFrames * bytesPerFrame_;
 493:   int32_t bufferSize = dataSize / sizeof(int16_t);
 494:
 495:   // Use new(std::nothrow) to avoid throwing exception
 496:   int16_t *buffer = new (std::nothrow) int16_t[bufferSize];
 497:   if (!buffer) {
 498:     LOG_ERROR("AudioPlayer::write: Failed to allocate buffer (%d bytes)",
 499:               dataSize);
 500:     return 0; // Memory allocation failed, return
 501:   }
 502:
 503:   memcpy(buffer, data, dataSize);
 504:
 505:   // Audio effect reactive: call callback to pass PCM data for spectrum analysis
 506:   {
 507:     std::lock_guard<std::mutex> callbackLock(audioCallbackMutex_);
 508:     if (audioDataCallback_) {
 509:       audioDataCallback_(static_cast<const int16_t*>(data), numFrames, sampleRate_);
 510:     }
 511:   }
 512:
 513:   // Add to queue (including frame count info)
 514:   {
 515:     std::lock_guard<std::mutex> lock(queueMutex_);
 516:     audioQueue_.push(AudioQueueItem(buffer, numFrames));
 517:
 518:   }
 519:   queueCondition_.notify_one();
 520:
 521:   // 防爆音：flush() 后的首次成功入队触发自动淡入，从 0 平滑爬升到目标音量。
 522:   if (pendingFadeInAfterFlush_.exchange(false, std::memory_order_acq_rel)) {
 523:     int32_t durMs = pendingFadeInDurationMs_.load(std::memory_order_relaxed);
 524:     if (durMs < 30) durMs = 30;
 525:     // 从当前（刚被 flush 归零的）currentVolume_=0 爬升到 volume_
 526:     fadeToVolume(volume_.load(), durMs);
 527:     LOG_DEBUG("AudioPlayer::write: triggered auto fade-in after flush, duration=%dms target=%.2f",
 528:               durMs, volume_.load());
 529:   }
 530:
 531:   return numFrames;
 532: #else
 533:   return 0;
 534: #endif
 535: }

 557: void AudioPlayer::fadeToVolume(float targetVolume, int32_t durationMs) {
 558: #ifdef __ANDROID__
 559:   float clampedTarget = std::max(0.0f, std::min(1.0f, targetVolume));
 560:   targetVolume_ = clampedTarget;
 561:
 562:   // Calculate volume change step per frame
 563:   // Assuming sample rate is 48000Hz, each callback ~256 frames, so ~187 callbacks/sec
 564:   // More precise calculation: based on actual sample rate and buffer size
 565:   float framesPerCallback = 256.0f; // AAudio typical value
 566:   float callbacksPerSecond = sampleRate_ / framesPerCallback;
 567:   float totalCallbacks = (durationMs / 1000.0f) * callbacksPerSecond;
 568:
 569:   if (totalCallbacks < 1.0f) {
 570:     totalCallbacks = 1.0f;
 571:   }
 572:
 573:   float currentVol = currentVolume_.load();
 574:   float volumeDiff = clampedTarget - currentVol;
 575:   volumeStep_ = volumeDiff / totalCallbacks;
 576:   fadeComplete_ = false;
 577:
 578:   LOG_DEBUG(
 579:       "AudioPlayer::fadeToVolume: from %.2f to %.2f, step=%.4f, duration=%dms",
 580:       currentVol, clampedTarget, volumeStep_.load(), durationMs);
 581: #endif
 582: }

 691: double AudioPlayer::estimatedOutputDelaySeconds() const {
 692: #ifdef __ANDROID__
 693:   if (!stream_ || sampleRate_ <= 0) return 0.0;
 694:
 695:   const int64_t framesWritten = AAudioStream_getFramesWritten(stream_);
 696:   if (framesWritten <= 0) return 0.0;
 697:
 698:   // 优先：AAudioStream_getTimestamp 直接给出"该帧将于 hwTimeNs 出声"。
 699:   //   delay = (framesWritten - hwFramePos) / sampleRate
 700:   // 该值天然包含 AAudio buffer + HAL + DSP + HDMI 全链路延迟。
 701:   // RK3588 SHARED 模式下该 API 经常 ERROR_INVALID_STATE，故必须有 fallback。
 702:   double instantSec = -1.0;
 703:   int64_t hwFramePos = 0;
 704:   int64_t hwTimeNs = 0;
 705:   aaudio_result_t res = AAudioStream_getTimestamp(stream_, CLOCK_MONOTONIC,
 706:                                                   &hwFramePos, &hwTimeNs);
 707:   if (res == AAUDIO_OK && hwFramePos > 0 && hwTimeNs > 0 &&
 708:       framesWritten >= hwFramePos) {
 709:     int64_t framesAhead = framesWritten - hwFramePos;
 710:     instantSec = static_cast<double>(framesAhead) /
 711:                  static_cast<double>(sampleRate_);
 712:   } else {
 713:     // Fallback：framesWritten - framesRead 只覆盖 AAudio app-side buffer，
 714:     // HAL + DSP + HDMI sink + TV 音频处理通常另加 150-300ms，
 715:     // 经 RK3588 HDMI 实测：inFlight ~74ms + 经验余量 0.18s ≈ 实际可闻延迟 250ms。
 716:     const int64_t framesRead = AAudioStream_getFramesRead(stream_);
 717:     int64_t inFlight = framesWritten - (framesRead > 0 ? framesRead : 0);
 718:     if (inFlight < 0) inFlight = 0;
 719:     instantSec = static_cast<double>(inFlight) /
 720:                  static_cast<double>(sampleRate_) + 0.18;
 721:   }
 722:
 723:   if (instantSec < 0.0) instantSec = 0.0;
 724:   if (instantSec > 1.0) instantSec = 1.0;  // 安全 clamp
 725:
 726:   // EMA 平滑：首次直接采用，后续 alpha=0.1
 727:   int64_t prevNs = smoothedOutputDelayNs_.load(std::memory_order_relaxed);
 728:   double smoothedSec;
 729:   if (prevNs <= 0) {
 730:     smoothedSec = instantSec;
 731:   } else {
 732:     const double prevSec = static_cast<double>(prevNs) / 1e9;
 733:     smoothedSec = prevSec * 0.9 + instantSec * 0.1;
 734:   }
 735:   smoothedOutputDelayNs_.store(static_cast<int64_t>(smoothedSec * 1e9),
 736:                                std::memory_order_relaxed);
 737:   return smoothedSec;
 738: #else
 739:   return 0.0;
 740: #endif
 741: }

 743: int32_t AudioPlayer::getPendingFrames() const {
 744: #ifdef __ANDROID__
 745:   std::lock_guard<std::mutex> lock(const_cast<AudioPlayer*>(this)->queueMutex_);
 746:   int32_t totalFrames = 0;
 747:
 748:   // Copy a queue or traverse queue to calculate total frames
 749:   // Note: this is frequently called in decodeThread, performance is important
 750:   // Here we traverse the queue
 751:   auto tempQueue = audioQueue_;
 752:   while(!tempQueue.empty()) {
 753:     const auto& item = tempQueue.front();
 754:     // Calculate remaining unplayed frames for this item
 755:     totalFrames += item.numFrames - (item.usedSamples / channelCount_);
 756:     tempQueue.pop();
 757:   }
 758:   return totalFrames;
 759: #else
 760:   return 0;
 761: #endif
 762: }

 801:
 802: aaudio_data_callback_result_t AudioPlayer::onAudioReady(void *audioData,
 803:                                                         int32_t numFrames) {
 804: #ifdef __ANDROID__
 805:   // 棣栧厛妫€鏌ユ槸鍚︽鍦ㄥ叧闂紙鏈€楂樹紭鍏堢骇锛?
 806:   if (shuttingDown_.load(std::memory_order_acquire)) {
 807:     if (audioData && numFrames > 0 && bytesPerFrame_ > 0) {
 808:       memset(audioData, 0, static_cast<size_t>(numFrames) * bytesPerFrame_);
 809:     }
 810:     return AAUDIO_CALLBACK_RESULT_CONTINUE;
 811:   }
 812:
 813:   // Basic parameter validation to guard against buggy drivers returning
 814:   // invalid callback arguments which could cause out-of-bounds writes.
 815:   if (!audioData || numFrames <= 0) {
 816:     return AAUDIO_CALLBACK_RESULT_CONTINUE;
 817:   }
 818:
 819:   if (channelCount_ <= 0 || bytesPerFrame_ <= 0) {
 820:     return AAUDIO_CALLBACK_RESULT_CONTINUE;
 821:   }
 822:
 823:   // Clamp excessively large numFrames to avoid integer overflow when
 824:   // multiplying by channelCount_ or bytesPerFrame_.
 825:   const int32_t MAX_FRAMES_PER_CALLBACK = 8192; // ~170ms at 48kHz
 826:   if (numFrames > MAX_FRAMES_PER_CALLBACK) {
 827:     static int warnCount = 0;
 828:     if (warnCount++ % 100 == 0) {
 829:       LOG_WARN(
 830:           "AudioPlayer::onAudioReady: numFrames too large (%d), clamped to %d",
 831:           numFrames, MAX_FRAMES_PER_CALLBACK);
 832:     }
 833:     numFrames = MAX_FRAMES_PER_CALLBACK;
 834:   }
 835:
 836:   int64_t totalSamples64 =
 837:       static_cast<int64_t>(numFrames) * static_cast<int64_t>(channelCount_);
 838:   if (totalSamples64 <= 0 ||
 839:       totalSamples64 > static_cast<int64_t>(std::numeric_limits<int32_t>::max())) {
 840:     memset(audioData, 0, static_cast<size_t>(numFrames) * bytesPerFrame_);
 841:     return AAUDIO_CALLBACK_RESULT_CONTINUE;
 842:   }
 843:
 844:   int32_t totalSamples = static_cast<int32_t>(totalSamples64);
 845:   int16_t *output = static_cast<int16_t *>(audioData);
 846:
 847:   memset(output, 0, static_cast<size_t>(numFrames) * bytesPerFrame_);
 848:
 849:   // Process fade in/out, update current volume
 850:   float currentVol = currentVolume_.load();
 851:   float targetVol = targetVolume_.load();
 852:   float step = volumeStep_.load();
 853:   bool fading = !fadeComplete_.load();
 854:
 855:   if (fading) {
 856:     // Calculate new current volume
 857:     if (step > 0) {
 858:       // Fade in
 859:       currentVol += step;
 860:       if (currentVol >= targetVol) {
 861:         currentVol = targetVol;
 862:         fadeComplete_ = true;
 863:       }
 864:     } else if (step < 0) {
 865:       // Fade out
 866:       currentVol += step;
 867:       if (currentVol <= targetVol) {
 868:         currentVol = targetVol;
 869:         fadeComplete_ = true;
 870:       }
 871:     } else {
 872:       fadeComplete_ = true;
 873:     }
 874:     currentVolume_ = currentVol;
 875:   }
 876:
 877:   // Read data from queue
 878:   std::unique_lock<std::mutex> lock(queueMutex_);
 879:
 880:   int32_t samplesWritten = 0;
 881:   size_t queueSizeBefore = audioQueue_.size();
 882:   int32_t samplesRemaining = totalSamples;
 883:   int32_t outputOffset = 0;
 884:
 885:   while (samplesRemaining > 0 && !audioQueue_.empty()) {
 886:     AudioQueueItem &item = audioQueue_.front();
 887:     int32_t itemRemainingSamples = item.getRemainingSamples(channelCount_);
 888:
 889:     if (itemRemainingSamples <= 0) {
 890:       // Queue item fully used, delete it
 891:       if (item.data) {
 892:           delete[] item.data;
 893:           item.data = nullptr;
 894:       }
 895:       audioQueue_.pop();
 896:       continue;
 897:     }
 898:
 899:     // Calculate samples to write this time
 900:     int32_t samplesToWrite = std::min(samplesRemaining, itemRemainingSamples);
 901:
 902:     // Apply volume and copy data (using smoothly transitioning current volume)
 903:     int32_t itemDataOffset = item.usedSamples;
 904:
 905:     if (fading) {
 906:       // Fade mode: adjust volume progressively for each sample
 907:       float volStart = currentVol - step; // Volume at start of this frame
 908:       float volEnd = currentVol;          // Volume at end of this frame
 909:       float volDelta = (volEnd - volStart) / samplesToWrite;
 910:       float sampleVol = volStart;
 911:
 912:       for (int32_t i = 0; i < samplesToWrite; i++) {
 913:         output[outputOffset + i] =
 914:             static_cast<int16_t>(item.data[itemDataOffset + i] * sampleVol);
 915:         sampleVol += volDelta;
 916:       }
 917:     } else {
 918:       // Normal mode: use fixed volume
 919:       for (int32_t i = 0; i < samplesToWrite; i++) {
 920:         output[outputOffset + i] =
 921:             static_cast<int16_t>(item.data[itemDataOffset + i] * currentVol);
 922:       }
 923:     }
 924:
 925:     // Update used sample count
 926:     item.usedSamples += samplesToWrite;
 927:     samplesWritten += samplesToWrite;
 928:     outputOffset += samplesToWrite;
 929:     samplesRemaining -= samplesToWrite;
 930:
 931:     // If queue item data fully used, delete it
 932:     if (item.usedSamples >= item.numFrames * channelCount_) {
 933:       if (item.data) {
 934:         delete[] item.data;
 935:         item.data = nullptr;
 936:       }
 937:       audioQueue_.pop();
 938:     }
 939:   }
 940:
 941:
 942:   lock.unlock();
 943:   queueCondition_.notify_one();
 944:
 945:   // Apply channel mode processing (after volume processing)
 946:   // Only stereo (2 channels) needs processing
 947:   if (channelCount_ == 2 && samplesWritten > 0) {
 948:     AudioChannelMode mode = channelMode_.load();
 949:     if (mode == AudioChannelMode::LEFT) {
 950:       // Play left channel only: copy left channel to right channel
 951:       // Sample format: L0 R0 L1 R1 L2 R2 ...
 952:       int32_t numFramesWritten = samplesWritten / 2;
 953:       for (int32_t i = 0; i < numFramesWritten; i++) {
 954:         int16_t leftSample = output[i * 2];      // Left channel
 955:         output[i * 2 + 1] = leftSample;          // Copy to right channel
 956:       }
 957:     } else if (mode == AudioChannelMode::RIGHT) {
 958:       // Play right channel only: copy right channel to left channel
 959:       int32_t numFramesWritten = samplesWritten / 2;
 960:       for (int32_t i = 0; i < numFramesWritten; i++) {
 961:         int16_t rightSample = output[i * 2 + 1]; // Right channel
 962:         output[i * 2] = rightSample;             // Copy to left channel
 963:       }
 964:     }
 965:     // STEREO mode doesn't need processing, keep as is
 966:   }
 967:
 968:   // Debug: record callback state
 969:   // If queue is large but written samples is small, consumption may be too slow (only warn on first detection)
 970:   static bool slowConsumptionWarned = false;
 971:   if (!slowConsumptionWarned && queueSizeBefore > 50 && samplesWritten < totalSamples / 2) {
 972:     LOG_WARN("AudioPlayer::onAudioReady: Large queue (%zu items) but slow "
 973:              "consumption (%d/%d samples)",
 974:              queueSizeBefore, samplesWritten, totalSamples);
 975:     slowConsumptionWarned = true;
 976:   }
 977:
 978:   return AAUDIO_CALLBACK_RESULT_CONTINUE;
 979: #else
 980:   return AAUDIO_CALLBACK_RESULT_STOP;
 981: #endif
 982: }

1008:                                           int32_t format,
1009:                                           int32_t deviceId) {
1010:     std::lock_guard<std::mutex> lock(mutex_);
1011:
1012:     if (initialized_ && audioPlayer_) {
1013:         const bool sameParams =
1014:             audioPlayer_->getSampleRate() == sampleRate &&
